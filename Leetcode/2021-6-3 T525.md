class Solution {
public:
    int findMaxLength(vector<int>& nums) {
        //使用map记录0，1数量相同，0的个数-1的个数=0.将0看成-1，计算最长连续子数组的和=0.与560题相似，
        map<int,int> mp;
        int pre=0,ans=0,preIndex;
        //计算前缀和
        mp[0]=-1;
        for(int i=0;i<nums.size();i++){
            if(nums[i]==1){
                pre+=1;
            }else{
                pre-=1;
            }
            if(mp.count(pre)){
                preIndex=mp[pre];
                ans=max(ans,i-preIndex);
            }
            else{
                mp[pre]=i;
            }
        }
        return ans;
    }
};
