//暴力...
class Solution {
public:
    bool wordPattern(string pattern, string s) {
        vector<string> arr;
        map<char,string> mp;
        string x="";
        int j=0;
        bool flag=true;
        for(int i=0;i<s.length();i++){
            if(s.at(i)!=' '){
                x=x+s.at(i);
            }else{
                arr.push_back(x);
                x="";
            }
        }
        arr.push_back(x);
        if(pattern.length()!=arr.size())return false;
        for(char c:pattern){
            if(mp.count(c)){
                if(arr[j]!=mp[c]){
                    flag=false;
                    break;
                }
            }else{
                for(auto& a:mp){
                    if(a.second==arr[j]) return false;
                }
                mp[c]=arr[j];
            }
            j++;
        }
        return flag;
    }
};
