```c++
class Solution {
public:
    bool isPowerOfFour(int n) {
        bool flag=true;
        if(n<0)return false;
        for(int i=0;i<=n;i++){
            long tmp=pow(4,i);
            if(tmp>n){
                return false;
            }else if(tmp==n){
                break;
            }
        }
        return flag;
    }
};
```
