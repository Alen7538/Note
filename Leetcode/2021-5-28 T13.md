//水题暴力...关键就是读罗马数字规则，从左往右
class Solution {
public:
    int romanToInt(string s) {
        int ans=0;
        map<char,int> mp;
        mp['I']=1;
        mp['V']=5;
        mp['X']=10;
        mp['L']=50;
        mp['C']=100;
        mp['D']=500;
        mp['M']=1000;
        reverse(s.begin(),s.end());
        int index=mp[s.at(0)];
        //cout<<index<<endl;
        ans+=index;
        for(int i=1;i<s.length();i++){
            //cout<<s.at(i);
            int temp=mp[s.at(i)];
            if(temp<index){
                ans-=temp;
            }else{
                ans+=temp;
            }
            index=temp;
        }
        return ans;
    }
};
