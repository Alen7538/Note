// The API isBadVersion is defined for you.
// bool isBadVersion(int version);

//查找
class Solution {
public:
    int firstBadVersion(int n) {
        long ans;
        long left=1,right=n;
        while(left<=right){
            long mid=(right+left)/2;
            if(isBadVersion(mid)){
                if(!isBadVersion(mid-1)){
                    ans=mid;
                    break;
                }
                right=mid-1;
            }else{
                if(isBadVersion(mid+1)){
                    ans=mid+1;
                    break;
                }
                left=mid+1;
            }
        }
        return ans;
    }
};
